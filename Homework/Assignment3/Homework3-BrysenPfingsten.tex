\textbf{c) Work}
\paragraph{Policy.} Non-preemptive FCFS (First-Come, First-Served).  
Let $A_i$ be arrival, $B_i$ be burst, $S_i$ start, $C_i$ completion, $T_i$ turnaround, $W_i$ waiting, $R_i$ response.  
For FCFS:
\[
\begin{aligned}
S_i &= \max\{C_{\,\text{prev}},\,A_i\},\qquad \\
C_i &= S_i + B_i,\\
T_i &= C_i - A_i,\qquad \\
W_i &= T_i - B_i,\qquad \\
R_i &= S_i - A_i\quad(\text{for FCFS, }R_i=W_i). \\
\end{aligned}
\]

\subsection*{Determine execution order (by arrival)}
Sorted by arrival time: \fbox{$P_3 \to P_1 \to P_6 \to P_2 \to P_5 \to P_4$}.

\subsection*{Compute $S_i$ and $C_i$}
\[
\begin{aligned}
&\textbf{For }P_3:\quad
S_{3}=\max\{0,0\}=0,\quad
C_{3}=0+15=15.\\[4pt]
&\textbf{For }P_1:\quad
S_{1}=\max\{C_3, A_1\}=\max\{15,2\}=15,\quad
C_{1}=15+20=35.\\[4pt]
&\textbf{For }P_6:\quad
S_{6}=\max\{C_1, A_6\}=\max\{35,3\}=35,\quad
C_{6}=35+16=51.\\[4pt]
&\textbf{For }P_2:\quad
S_{2}=\max\{C_6, A_2\}=\max\{51,5\}=51,\quad
C_{2}=51+14=65.\\[4pt]
&\textbf{For }P_5:\quad
S_{5}=\max\{C_2, A_5\}=\max\{65,6\}=65,\quad
C_{5}=65+24=89.\\[4pt]
&\textbf{For }P_4:\quad
S_{4}=\max\{C_5, A_4\}=\max\{89,8\}=89,\quad
C_{4}=89+11=100.
\end{aligned}
\]

\subsection*{Per-process calculations}
\[
\begin{aligned}
\textbf{P}_1:\;&

\textbf{P}_4:\;&
T_4=C_4-A_4=100-8=92,\quad
W_4=T_4-B_4=92-11=81,\quad
R_4=S_4-A_4=89-8=81.\\[4pt]
\textbf{P}_5:\;&
T_5=C_5-A_5=89-6=83,\quad
W_5=T_5-B_5=83-24=59,\quad

\textbf{P}_6:\;&

W_6=T_6-B_6=48-16=32,\quad

\end{aligned}
\]
